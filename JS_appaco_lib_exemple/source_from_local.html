get all words from a local dectionary




html code :

<input type="file" id="file"><br/>
<input id="search" type="text"/>
<input type="button" id="next_result_button" value="next">

JavaScript code:

var file_input_id = 'file';
var fileInput = document.querySelector('#'+file_input_id);

fileInput.onchange = function() 
{
	var reader = new FileReader();
 
	reader.onload = function() 
	{	
		options.doSort = true;
		options.item_ID_OnClick_nextResult='next_result_button';
		options.score_update_unit = 0;
		
		approximate_autoComplete('search',reader.result);
	};
 
	reader.readAsText(fileInput.files[0],"UTF-8");
};
